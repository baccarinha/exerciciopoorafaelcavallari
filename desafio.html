Saque usando a Menor Nota

Código em Python

valor = int(input("Digite o valor para saque: "))

if valor <= 0:
    print("Valor inválido.")
elif valor == 1 or valor == 3:
    print("Não é possível sacar esse valor com as cédulas disponíveis.")
else:
    notas = [200, 100, 50, 20, 10, 5, 2]
    
    menor_nota = notas[-1]

    quantidade = valor // menor_nota

    print("Cédulas entregues:")
    print(f"{quantidade} nota(s) de R$ {menor_nota}")
