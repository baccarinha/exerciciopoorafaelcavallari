Código em Python:

# Notas disponíveis
notas = [200, 100, 50, 20, 10, 5, 2]

valor = int(input("Digite o valor para saque: R$ "))

if valor < 2:
    print("Valor inválido para saque.")
else:
    print(f"\nPara sacar R$ {valor}, serão necessárias:")

    for nota in notas:
        quantidade = valor // nota
        if quantidade > 0:
            print(f"{quantidade} nota(s) de R$ {nota}")
            valor = valor % nota

    if valor != 0:
        print("\nNão é possível sacar o valor exato com as notas disponíveis.")
